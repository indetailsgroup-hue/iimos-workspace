drawer-calculation.ts 
interface DrawerParams {
  openingWidth: number;
  runnerType: 'MOVENTO' | 'TANDEM' | 'Standard';
}

function calculateDrawerWidth(
  params: DrawerParams
): number {
  const clearance =
    params.runnerType === 'MOVENTO' ||
    params.runnerType === 'TANDEM'
      ? 42  // Blum full-extension
      : 26; // Standard runners

  return params.openingWidth - clearance;
}
